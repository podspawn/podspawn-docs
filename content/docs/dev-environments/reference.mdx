Complete reference for all Podfile fields. For a tutorial introduction, see [Getting Started](/docs/dev-environments/quickstart).

## Full schema

```yaml
# Base image (required unless extends is set)
base: ubuntu:24.04

# Inherit from another Podfile
extends: ubuntu-dev  # registry name, local path, or GitHub URL

# Session name override (default: derived from directory name + hash)
name: my-project

# Packages to install
packages:
  - go@1.24
  - nodejs@22
  - ripgrep

# Default shell
shell: /bin/bash  # default

# Dotfiles repository
dotfiles:
  repo: github.com/user/dotfiles
  install: ./install.sh

# Repositories to clone inside the container
repos:
  - url: github.com/org/repo
    path: /workspace/repo
    branch: main  # default

# Environment variables
env:
  KEY: value
  DYNAMIC: "${PODSPAWN_USER}"  # expanded at runtime

# Companion services
services:
  - name: postgres
    image: postgres:16
    ports: [5432]
    env:
      POSTGRES_PASSWORD: devpass
    volumes:
      - pgdata:/var/lib/postgresql/data

# Port forwarding
ports:
  expose: [3000, 8080]
  strategy: auto  # auto | manual | expose

# Resource limits
resources:
  cpus: 2.0
  memory: 4g

# Lifecycle hooks
on_create: |    # runs once on first container start
  npm install
on_start: |     # runs every time you attach
  echo "ready"

# Additional Dockerfile RUN commands
extra_commands:
  - "apt-get install -y custom-tool"

# How the host directory is available inside the container
mount: bind  # bind | copy | none

# Session lifecycle mode
mode: grace-period  # grace-period | destroy-on-disconnect | persistent

# Mount target inside container (default: /workspace/<dirname>)
workspace: /workspace/myapp
```

    <TypeTable type={{
      base: { type: 'string', description: 'Base Docker image. Required unless extends is set.', default: '—' },
      extends: { type: 'string', description: 'Inherit from another Podfile. Registry name, local path, or GitHub URL.', default: '—' },

      mount: { type: '"bind" | "copy" | "none"', description: 'How the host directory is available inside the container.', default: 'bind' },

      workspace: { type: 'string', description: 'Mount target inside container. Must be absolute.', default: '/workspace/<dir>' },
    }} />

    Merge behavior: all scalar fields use child-wins (child value replaces base if non-empty).
  </Accordion>
  <Accordion title="Packages and extra commands">
    <TypeTable type={{
      packages: { type: 'string[]', description: 'Packages to install. Supports version pinning (go@1.24).', default: '[]' },
      extra_commands: { type: 'string[]', description: 'Additional Dockerfile RUN commands.', default: '[]' },
    }} />

    Merge behavior: child appends to base, duplicates removed. `"!item"` removes from base.
  </Accordion>
  <Accordion title="Environment variables">
    <TypeTable type={{

    Merge behavior: maps are merged, child wins on key conflict. Empty value `""` removes the key from the base.
  </Accordion>
  <Accordion title="Services">

      'services[].image': { type: 'string', description: 'Docker image for the service.', default: '—' },
      'services[].ports': { type: 'number[]', description: 'Ports the service exposes on the shared network.', default: '[]' },

      'services[].volumes': { type: 'string[]', description: 'Named volumes in name:/path format.', default: '[]' },
    }} />

    Merge behavior: same-name services are replaced by child. Different names are appended. `"!name"` removes.
  </Accordion>
  <Accordion title="Ports">
    <TypeTable type={{
      'ports.expose': { type: 'number[]', description: 'Ports forwarded from container to host.', default: '[]' },
      'ports.strategy': { type: '"auto" | "manual" | "expose"', description: 'How port conflicts are handled.', default: 'auto' },
    }} />

    Merge behavior: `ports.expose` appends and deduplicates (no bang-replace support). `ports.strategy` uses child-wins.
  </Accordion>
  <Accordion title="Resources">
    <TypeTable type={{
      'resources.cpus': { type: 'float', description: 'CPU limit for the container.', default: '—' },
      'resources.memory': { type: 'string', description: 'Memory limit (e.g., "4g", "512m").', default: '—' },
    }} />

    Merge behavior: per-field, child wins if set (cpus > 0, memory non-empty).
  </Accordion>
  <Accordion title="Lifecycle hooks">
    <TypeTable type={{
      on_create: { type: 'string', description: 'Shell commands run once on first container start. Heavy setup goes here.', default: '—' },
      on_start: { type: 'string', description: 'Shell commands run every time you attach.', default: '—' },
    }} />

    Merge behavior: concatenated (base runs first, then child).
  </Accordion>
  <Accordion title="Dotfiles and repos">
    <TypeTable type={{
      'dotfiles.repo': { type: 'string', description: 'Git URL for your dotfiles repository.', default: '—' },

    Merge behavior: `dotfiles` is replaced entirely if child sets it. `repos` appends; same-URL entries have branch/path overridden by child.

## Bang-replace fields

These fields support the `!` suffix to fully replace instead of merge:

`packages!`, `env!`, `services!`, `repos!`, `extra_commands!`, `on_create!`, `on_start!`

<Callout type="info">
`ports.expose` does not support bang-replace. Ports are always merged.
</Callout>

## Variable expansion

Environment variables support these built-in variables:

| Variable | Expands to |
|----------|------------|
| `${PODSPAWN_USER}` | The authenticated username |
| `${PODSPAWN_PROJECT}` | The project/session name |

## Package version syntax

| Package | Syntax | Install method |
|---------|--------|---------------|
| Go | `go@1.24` | Official tarball |
| Node.js | `nodejs@22` | NodeSource PPA |
| Python | `python@3.13` | deadsnakes PPA |
| Rust | `rust` | rustup |
| Others | `ripgrep`, `jq` | apt-get |

## Validation rules

<Accordions>
  <Accordion title="All validation rules">
    - `base` is required unless `extends` is set
    - `shell` must be an absolute path
    - `mount` must be `bind`, `copy`, or `none`
    - `mode` must be `grace-period`, `destroy-on-disconnect`, or `persistent`
    - `ports.strategy` must be `auto`, `manual`, or `expose`
    - `workspace` must be an absolute path
    - Repo `url` is required; repo paths must be absolute
    - Service `name` and `image` are required
    - Service names must be unique
    - `"!item"` removal syntax only valid when `extends` is set
  </Accordion>
</Accordions>
